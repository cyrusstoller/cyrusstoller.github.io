Move all files in a directory into their own directories.

for f in *
do
  y=${f/.*/}
  mkdir "$y"
  mv "$f" "$y"/
done

Change all file names in a directory to be lowercase.

for SRC in `find . -depth`
do
  DST=`dirname "${SRC}"`/`basename "${SRC}" | tr '[A-Z]' '[a-z]'`
  if [ "${SRC}" != "${DST}" ]
  then
    mv "${SRC}" "${DST}"
  fi
done
